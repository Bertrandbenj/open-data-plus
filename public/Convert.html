import scala.util.matching.Regex
import spark.implicits._
case class Quad(subject: String, predicate: String, objet: String, graph: String)
val uri = """<[^\s]+>"""
val graph = """("+uri+").*$""".r
val blankNode = "_:(?:[A-Za-z][A-Za-z0-9\\-_]*)"
val langOrType = "(?:@(?:[a-z]+[\\-A-Za-z0-9]*)|\\^\\^<(?:[^>]+)>)?"
val string = "\\\"(?:(?:\\\"|[^\"])*)\\\""
val obj = "(<[^\\s]+>|_:(?:[A-Za-z][A-Za-z0-9\\-_]*)|\\\"(?:(?:\\\"|[^\"])*)\\\"(?:@(?:[a-z]+[\\-A-Za-z0-9]*)|\\^\\^<(?:[^>]+)>)?)".r
//var pattern = """^(" + uri +"|" + blankNode +")\\s+("+uri+")\\s+("+obj+")\\s+("+uri+").*$""".r
var pattern = "^(<[^\\s]+>|_:(?:[A-Za-z][A-Za-z0-9\\-_]*))\\s+(<[^\\s]+>)\\s+(.*)\\s+(<[^\\s]+>).*$".r
var originalPattern =
"^(<[^\\s]+>|_:(?:[A-Za-z][A-Za-z0-9\\-_]*))\\s+(<[^\\s]+>)\\s+(<[^\\s]+>|_:(?:[A-Za-z][A-Za-z0-9\\-_]*)|\\\"(?:(?:\\\"|[^\"])*)\\\"
(?:@(?:[a-z]+[\\-A-Za-z0-9]*)|\\^\\^<(?:[^>]+)>)?)\\s+(<[^\\s]+>).*$".r
def prefixIt(s:String) : String = {
val prefixes = Map( "http://www.w3.org/1999/02/22-rdf-syntax-ns#" -> "rdf_",
"http://www.w3.org/2006/vcard/ns#" -> "vcard_",
"http://schema.org/" -> "sch_")
var res = s.replaceFirst("<","").replaceFirst(">","")
for ((k,v) <- prefixes)
res = res.replaceAll(k,v)
return res
}
def buildQuad(s:String) : Quad = {
pattern.findFirstMatchIn(s) match {
case Some(m) =>
obj.findFirstMatchIn(m.group(3)) match {
case Some(o) =>
Quad(m.group(1), prefixIt(m.group(2)), prefixIt(o.group(1)), m.group(4) )
case None =>
println("Problem parsing object "+m.group(3));
null
}
case None =>
println("Problem parsing pattern for String :\n"+s)
null
}
}
spark.read
.textFile("ofInterest.gz/*txt")
.filter(line => { if (line.length > 3000) println(line ); line.length < 3000} ) //fancy regex overflow
.map(buildQuad(_))
//.show
.filter(q => q != null && q.graph != null) // unfortonately spark map isn't filtering null
.withColumnRenamed("objet","object") // Because .. ho well
.write
.mode("overwrite")
.bucketBy(1, "graph")
.sortBy("graph", "subject", "predicate")
.saveAsTable("openNantes");
// CHECK THE RESULT
spark.sql("SELECT * FROM openNantes")
.show();

# File conversion
XXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXX

File conversion

In order to convert the file to a column based structured parquet file, we match a regexp, the unzipping (.gz) is handled natively by spark, the regexp being greedy and buggy we filter big lines (> 2000 char) mostly articleBody vectors or the like. we loose some information there but avoid as well a lot of noise.

REGEX

The regular expression capture 4 groups Some filtering needs to be done afterwards.

Writing result

Results get written to parquet by the cluster without going through driver program.
